"""Веб-інтерфейс: список сайтів -> кваліфікація по кожному.""" import os, sys, concurrent.futures from flask import Flask, render_template, request, jsonify sys.path.insert(0, os.path.dirname(os.path.dirname(os.path.abspath(__file__)))) from analyzer import qualify, config app = Flask(__name__) MAX_DOMAINS = int(os.getenv("MAX_DOMAINS", "100")) WORKERS = int(os.getenv("WORKERS", "6")) def _parse_domains(raw: str): items = [] for line in raw.replace(",", "\n").splitlines(): d = line.strip().lower() if not d: continue d = d.replace("https://", "").replace("http://", "").strip("/ ") if d and d not in items: items.append(d) return items[:MAX_DOMAINS] def _run(domains, do_onpage): results = [] with concurrent.futures.ThreadPoolExecutor(max_workers=WORKERS) as ex: futs = {ex.submit(_safe_qualify, d, do_onpage): d for d in domains} for fut in concurrent.futures.as_completed(futs): results.append(fut.result()) results.sort(key=lambda r: r.get("score", 0), reverse=True) return results def _safe_qualify(domain, do_onpage): try: return qualify.qualify(domain, do_onpage=do_onpage) except Exception as e: return {"domain": domain, "verdict": "ПОМИЛКА", "color": "gray", "score": -1, "error": str(e)[:200], "reasons": [], "metrics": {}, "dotisk_queries": []} @app.route("/") def = request.form.get("onpage") == "on" domains = _parse_domains(raw) results = _run(domains, do_onpage) if domains else [] return render_template("results.html", results=results, cfg=config, do_onpage=do_onpage) @app.route("/api/analyze", methods=["POST"]) def api_analyze(): data = request.get_json(force=True, silent=True) or {} domains = _parse_domains("\n".join(data.get("domains", [])) if isinstance(data.get("domains"), list) else str(data.get("domains", ""))) do_onpage = bool(data.get("onpage", True)) return jsonify({"results": _run(domains, do_onpage) if domains else []})
